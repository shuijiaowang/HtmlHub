# HtmlHub 域名说明（四级用户页 + 三级控制台）
# - 控制台（Vue + /api）：https://htmlhub.lyyxy.top
# - 用户上传的静态页：   https://{slug}.htmlhub.lyyxy.top  （四级）
# 这样根域下的其它三级子域（如 blog.lyyxy.top）可留给别的项目，不与 HtmlHub 抢泛解析。
# DNS 建议：
#   - A/AAAA：htmlhub.lyyxy.top -> 服务器 IP
#   - 泛解析：*.htmlhub.lyyxy.top -> 同上（或与 htmlhub 同一条泛解析规则，视 DNS 商而定）
# SSL：证书需包含 htmlhub.lyyxy.top 与 *.htmlhub.lyyxy.top（或通配符 *.htmlhub.lyyxy.top 含父名，按证书策略配置）。

# ---------- 站点 1：控制台（静态前端 + 仅 /api 走 Go）----------
server
{
		listen 7788;
    server_name htmlhub.lyyxy.top 59.110.214.77;
    index index.php index.html index.htm default.php default.htm default.html;
    root /www/wwwroot/htmlhub/web;
    #CERT-APPLY-CHECK--START
    # 用于SSL证书申请时的文件验证相关配置 -- 请勿删除
    include /www/server/panel/vhost/nginx/well-known/59.110.214.77.conf;
    #CERT-APPLY-CHECK--END
    include /www/server/panel/vhost/nginx/extension/59.110.214.77/*.conf;

    #SSL-START SSL相关配置，请勿删除或修改下一行带注释的404规则
    #error_page 404/404.html;
    #SSL-END

    #ERROR-PAGE-START  错误页配置，可以注释、删除或修改
    error_page 404 /404.html;
    #error_page 502 /502.html;
    #ERROR-PAGE-END

    #PHP-INFO-START  PHP引用配置，请勿删除或修改
    include enable-php-00.conf;
    #PHP-INFO-END

    #REWRITE-START URL重写规则引用,修改后将导致面板设置的伪静态规则失效
    include /www/server/panel/vhost/rewrite/59.110.214.77.conf;
    #REWRITE-END
    # ---------- 接口：转发到本机 Go（端口与 main.go 一致）----------
    location ^~ /api/ {
        proxy_pass http://127.0.0.1:7789;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
    # ---------- 前端：Vue Router history ----------
    location / {
        try_files $uri $uri/ /index.html;
    }

    # 禁止访问的敏感文件
    location ~* (\.user.ini|\.htaccess|\.htpasswd|\.env.*|\.project|\.bashrc|\.bash_profile|\.bash_logout|\.DS_Store|\.gitignore|\.gitattributes|LICENSE|README\.md|CLAUDE\.md|CHANGELOG\.md|CHANGELOG|CONTRIBUTING\.md|TODO\.md|FAQ\.md|composer\.json|composer\.lock|package(-lock)?\.json|yarn\.lock|pnpm-lock\.yaml|\.\w+~|\.swp|\.swo|\.bak(up)?|\.old|\.tmp|\.temp|\.log|\.sql(\.gz)?|docker-compose\.yml|docker\.env|Dockerfile|\.csproj|\.sln|Cargo\.toml|Cargo\.lock|go\.mod|go\.sum|phpunit\.xml|phpunit\.xml|pom\.xml|build\.gradl|pyproject\.toml|requirements\.txt|application(-\w+)?\.(ya?ml|properties))$
    {
        return 404;
    }

    # 禁止访问的敏感目录
    location ~* /(\.git|\.svn|\.bzr|\.vscode|\.claude|\.idea|\.ssh|\.github|\.npm|\.yarn|\.pnpm|\.cache|\.husky|\.turbo|\.next|\.nuxt|node_modules|runtime)/ {
        return 404;
    }

    #一键申请SSL证书验证目录相关设置
    location ~ \.well-known{
        allow all;
    }

    #禁止在证书验证目录放入敏感文件
    if ( $uri ~ "^/\.well-known/.*\.(php|jsp|py|js|css|lua|ts|go|zip|tar\.gz|rar|7z|sql|bak)$" ) {
        return 403;
    }

    location ~ .*\.(gif|jpg|jpeg|png|bmp|swf)$
    {
        expires      30d;
        error_log /dev/null;
        access_log /dev/null;
    }

    location ~ .*\.(js|css)?$
    {
        expires      12h;
        error_log /dev/null;
        access_log /dev/null;
    }
    access_log  /www/wwwlogs/htmlhub-portal.log;
    error_log  /www/wwwlogs/htmlhub-portal.error.log;
}

# ---------- 站点 2：用户页四级域名，整站反代 Go（GET / 返回注入后的 HTML；/api 供同步脚本）----------
# 勿再使用 *.lyyxy.top 指向本配置，以免占用其它项目的三级域名。
server
{
		listen 7788;
    server_name *.htmlhub.lyyxy.top;
    include /www/server/panel/vhost/nginx/well-known/59.110.214.77.conf;

    location / {
        proxy_pass http://127.0.0.1:7789;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location ~ \.well-known{
        allow all;
    }

    access_log  /www/wwwlogs/htmlhub-userhtml.log;
    error_log  /www/wwwlogs/htmlhub-userhtml.error.log;
}

# ---------- 故障排查：四级子域打开却是 Vue「首页」或 /home 也是控制台 ----------
# 原因 A：请求仍进了「控制台」站点。location / 里 try_files 会回退到 index.html，Vue Router 再按路径渲染 /home，看起来像主页。
#   处理：在服务器执行 nginx -T | findstr /i "htmlhub"（Linux 用 grep）确认 todolist2.htmlhub.lyyxy.top 落在「站点 2」、整站 proxy_pass Go。
#   若仍进错站点：检查是否有其它 vhost 使用 server_name *.lyyxy.top 且 listen 同一端口并 default_server，或宝塔未合并本段配置。
# 原因 B：用户子域已反代 Go，但访问的是 /home。此前后端只注册了 GET /，现已增加 /home、/index.html 与 / 相同逻辑；仍建议分享链接用根路径 http://slug.htmlhub.lyyxy.top:7788/ 。
# 若 *.htmlhub.lyyxy.top 不生效，可改用正则（二选一）：
#   server_name ~^(?<sub>[^.]+)\.htmlhub\.lyyxy\.top$;
