### 前端

宝塔面板，网站，PHP项目，添加站点，域名：htmlhub.lyyxy.top(不需要加*号)，目录地址：/www/wwwroot/htmlhub/web，PHP版本：纯静态
前端，npm run build,把dist里的所有文件上传到目录：/www/wwwroot/htmlhub/web
nginx配置
htmlhub.lyyxy.top->http://127.0.0.1:7789/api
*htmlhub.lyyxy.top->http://127.0.0.1:7789
```
server
{
    listen 80;
    server_name htmlhub.lyyxy.top;
    # Vue 打包后的 dist 目录文件
    root /www/wwwroot/htmlhub/web;
    index index.html;

    # 宝塔SSL证书申请配置（保留）
    #CERT-APPLY-CHECK--START
    include /www/server/panel/vhost/nginx/well-known/htmlhub.lyyxy.top.conf;
    #CERT-APPLY-CHECK--END
    include /www/server/panel/vhost/nginx/extension/htmlhub.lyyxy.top/*.conf;

    #SSL-START
    #error_page 404/404.html;
    #SSL-END

    # 错误页
    error_page 404 /404.html;

    # 伪静态（保留宝塔配置）
    include /www/server/panel/vhost/rewrite/htmlhub.lyyxy.top.conf;

    # --------------------------
    # 核心：API 转发到 Go 后端 7789
    # --------------------------
    location ^~ /api/ {
        proxy_pass http://127.0.0.1:7789;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # --------------------------
    # Vue Router History 模式
    # --------------------------
    location / {
        try_files $uri $uri/ /index.html;
    }

    # --------------------------
    # 安全规则：禁止访问敏感文件/目录
    # --------------------------
    location ~* (\.user.ini|\.htaccess|\.env|\.git|README\.md|composer\..*|package\..*)$ {
        return 404;
    }
    location ~* /(\.git|\.idea|node_modules|runtime)/ {
        return 404;
    }

    # SSL 验证目录放行
    location ~ \.well-known{
        allow all;
    }
    if ( $uri ~ "^/\.well-known/.*\.(php|js|sql|bak)$" ) {
        return 403;
    }

    # 静态资源缓存
    location ~ .*\.(gif|jpg|jpeg|png|bmp|swf)$ {
        expires 30d;
        access_log off;
    }
    location ~ .*\.(js|css)$ {
        expires 12h;
        access_log off;
    }

    # 日志
    access_log  /www/wwwlogs/htmlhub.lyyxy.top.log;
    error_log  /www/wwwlogs/htmlhub.lyyxy.top.error.log;
}

# --------------------------
# 二级配置：泛子域名 *.htmlhub.lyyxy.top
# 全部转发到 Go 后端（用户HTML页面）
# --------------------------
server
{
    listen 80;
    server_name *.htmlhub.lyyxy.top;

    # 宝塔SSL验证配置
    include /www/server/panel/vhost/nginx/well-known/htmlhub.lyyxy.top.conf;

    # 全部转发到 Go 后端
    location / {
        proxy_pass http://127.0.0.1:7789;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # SSL 验证目录放行
    location ~ \.well-known{
        allow all;
    }

    # 日志
    access_log  /www/wwwlogs/htmlhub-user.log;
    error_log  /www/wwwlogs/htmlhub-user.error.log;
}
```
